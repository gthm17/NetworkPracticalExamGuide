TCP — Transmission Control Protocol Connection-Based

TCP is like a phone call — you establish a connection first, then talk, then hang up. Reliable, ordered delivery guaranteed.

How TCP Works — The Flow
1socket() — Both server & client create a socket (endpoint)
↓ Server only
2bind() — Server attaches socket to a port (3003)
3listen() — Server starts waiting for connections
4accept() — Server accepts incoming client
↓ Client only
5connect() — Client connects to server IP:port
↓ Both
6send() / recv() — Exchange data
7close() — Terminate connection
SERVER
socket() → bind() → listen() → accept()
↕3-Way Handshake (SYN, SYN-ACK, ACK)
CLIENT
socket() → connect()
↕send() / recv() both directions
Key Socket Args
AF_INET
IPv4 address family
SOCK_STREAM
TCP — stream based, reliable
INADDR_ANY
Accept connections on any interface
TCP Server SERVER

Server binds to port, listens, accepts client, then does recv → send.

// TCP SERVER — Networks Lab
#include <sys/socket.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <netinet/in.h>

int main() {
    int sock_desc, temp_sock_desc, k, len;
    char buf[100];
    struct sockaddr_in server_addr, client;

    // Step 1: Create socket
    sock_desc = socket(AF_INET, SOCK_STREAM, 0);
    if (sock_desc == -1)
        printf("Error in socket creation\n");

    // Step 2: Set server address
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(3003);

    // Step 3: Bind socket to address
    k = bind(sock_desc, (struct sockaddr*)&server_addr, sizeof(server_addr));
    if (k == -1)
        printf("Error in binding\n");

    // Step 4: Listen for connections
    k = listen(sock_desc, 5);
    if (k == -1)
        printf("Error in listening\n");

    // Step 5: Accept a client
    len = sizeof(client);
    temp_sock_desc = accept(sock_desc, (struct sockaddr*)&client, &len);
    if (temp_sock_desc == -1)
        printf("Error in accept\n");

    // Step 6: Receive message from client
    k = recv(temp_sock_desc, buf, 100, 0);
    printf("Message from client: %s\n", buf);

    // Step 7: Send reply
    printf("Enter data to send: ");
    fgets(buf, 100, stdin);
    send(temp_sock_desc, buf, 100, 0);

    close(temp_sock_desc);
    close(sock_desc);
    return 0;
}
🔑 REMEMBER
Server uses TWO sockets: sock_desc (listening) and temp_sock_desc (actual communication with client). accept() returns the second one.
TCP Client CLIENT

Client just creates socket, connects, then does send → recv. No bind/listen/accept needed!

// TCP CLIENT — Networks Lab
#include <sys/socket.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <netinet/in.h>

int main() {
    int sock_desc, k;
    char buf[100];
    struct sockaddr_in client;

    // Step 1: Create socket (SAME as server)
    sock_desc = socket(AF_INET, SOCK_STREAM, 0);
    if (sock_desc == -1)
        printf("Error in socket creation\n");

    // Step 2: Set server address to connect TO
    client.sin_family = AF_INET;
    client.sin_addr.s_addr = INADDR_ANY;
    client.sin_port = htons(3003); // SAME port as server

    // Step 3: Connect to server (replaces bind+listen+accept)
    k = connect(sock_desc, (struct sockaddr*)&client, sizeof(client));
    if (k == -1)
        printf("Error in connecting\n");

    // Step 4: Send message first (client initiates)
    printf("Enter data to send: ");
    fgets(buf, 100, stdin);
    send(sock_desc, buf, 100, 0);

    // Step 5: Receive reply from server
    k = recv(sock_desc, buf, 100, 0);
    printf("Message from server: %s\n", buf);

    close(sock_desc);
    return 0;
}

❌ REMOVE from server
bind() call • listen() call • accept() call • temp_sock_desc variable
➕ ADD in client
connect(sock_desc, (struct sockaddr*)&client, sizeof(client)) — this single line replaces ALL 3 server calls

StepSERVERCLIENT

3listen()❌ skip

💡 EXAM TIP
If asked "what does the client NOT need that server does?" → Answer: bind(), listen(), accept(). Client only needs socket() + connect().
UDP — User Datagram Protocol Connectionless

UDP is like sending a letter — no handshake, no guarantee. Faster but unreliable. Key difference from TCP: use SOCK_DGRAM and sendto()/recvfrom() instead of send()/recv().

TCP vs UDP Quick Diff
FeatureTCPUDP
Socket typeSOCK_STREAMSOCK_DGRAM
Connectionconnect()No connect
Sendsend()sendto()
Receiverecv()recvfrom()
Reliable✅ Yes❌ No
Server acceptaccept()No accept
UDP Flow
1socket(AF_INET, SOCK_DGRAM)
↓ Server only
2bind() — server binds to port
↓ No connection step!
3recvfrom() / sendto()
UDP Server SERVER
// UDP SERVER — Networks Lab
#include <stdio.h>
#include <sys/socket.h>
#include <stdlib.h>
#include <string.h>
#include <netinet/in.h>

int main() {
    int sock_id, k;
    char buf[100];
    struct sockaddr_in server, client;
    socklen_t len = sizeof(client);

    // Step 1: Create UDP socket (SOCK_DGRAM!)
    sock_id = socket(AF_INET, SOCK_DGRAM, 0);
    if (sock_id == -1)
        printf("Error in socket\n");

    // Step 2: Bind (server needs this)
    server.sin_family = AF_INET;
    server.sin_addr.s_addr = INADDR_ANY;
    server.sin_port = htons(3003);

    k = bind(sock_id, (struct sockaddr*)&server, sizeof(server));
    if (k == -1)
        printf("Error in binding\n");

    // Step 3: Receive from client (recvfrom, not recv!)
    k = recvfrom(sock_id, buf, 100, 0,
        (struct sockaddr*)&client, &len);
    printf("Got datagram: %s\n", buf);

    // Step 4: Send reply back
    printf("Enter reply: ");
    fgets(buf, 100, stdin);
    sendto(sock_id, buf, 100, 0,
        (struct sockaddr*)&client, len);

    close(sock_id);
    return 0;
}

    close(sock_id);

UDP — Copy-Paste Tricks
Server → Client Changes
✅ KEEP SAME
All headers • socket(AF_INET, SOCK_DGRAM, 0) • Port 3003 • sendto() / recvfrom() function signatures
❌ REMOVE from server to make client
The entire bind() block (client doesn't bind)
🔄 SWAP order
Server does recvfrom() first, client does sendto() first
💡 KEY TRICK — UDP vs TCP in one line
Change SOCK_STREAM → SOCK_DGRAM, change send/recv → sendto/recvfrom, remove listen() and accept() from server, remove connect() from client. That's literally the full TCP→UDP conversion.
Stop & Wait Protocol Flow Control

Send one packet → Wait for ACK → Send next. Simplest flow control protocol. Uses random error simulation with rand() % 4.

How It Works
1Sender sends packet with sequence number
2errorframe = rand() % 4 — simulates packet loss (0 = error)
3If no error: Receiver gets it, sends ACK back
4errorack = rand() % 4 — simulates ACK loss
5If ACK received: Sender moves to next packet
6If error: t-- (timeout countdown), retransmit
Key Variables
turn = 's' or 'r' (whose turn)
errorframe = 0 means error in packet
errorack = 0 means error in ACK
p.seq = sequence number
ak = acknowledgment number
frexp = expected frame number (receiver side)
The Core Logic
rand() % 4 == 0 means ~25% chance of error. This simulates real network unreliability. If errorframe ≠ 0, packet sent OK. If errorack ≠ 0, ACK sent OK.
Stop & Wait — Full Code
// STOP AND WAIT — Networks Lab
#include <stdio.h>
#include <stdlib.h>

struct frame {
    int info;
    int seq;
};

int ak;
int t = 5, k;
int disconnect = 0;
struct frame p;
char turn = 's';
int errorframe = 1, errorack = 1;

void sender();
void receiver();

void main() {
    p.info = 0;
    p.seq = 0;
    while (!disconnect) {
        sender();
        receiver();
    }
}

void sender() {
    static int flag = 0;
    if (turn == 's') {
        if (errorack == 0) {          // ACK was lost last time
            printf("SENDER: sent packet with seq NO:%d\n", p.seq);
            errorframe = rand() % 4;
            printf("%s\n", (errorframe == 0 ? "Error while sending Packet" : ""));
            turn = 'r';
        } else {
            if (flag == 1)
                printf("SENDER: Received ACK for packet %d\n", ak);
            if (p.seq == 5) {         // Sent all 5 packets
                disconnect = 1;
                return;
            }
            p.info = p.info + 1;
            p.seq = p.seq + 1;
            printf("SENDER: Sent packet with seq NO:%d\n", p.seq);
            errorframe = rand() % 4;
            printf("%s\n", (errorframe == 0 ? "Error While sending Packet" : ""));
            turn = 'r';
            flag = 1;
        }
    } else {
        t--;  // timeout countdown
    }
}

void receiver() {
    static int frexp = 1;  // expected frame
    if (turn == 'r') {
        if (errorframe != 0) {    // packet arrived OK
            if (p.seq == frexp) {  // correct sequence
                printf("RECEIVER: Received packet with seq %d\n", p.seq);
                ak = p.seq;
                frexp = frexp + 1;
                turn = 's';
                errorack = rand() % 4;
                printf("%s\n", (errorack == 0 ? "Error while sending ACK" : ""));
            } else {              // duplicate packet
                printf("RECEIVER: Duplicated packet with seq %d\n", p.seq);
                ak = frexp - 1;
                turn = 's';
                errorack = rand() % 4;
                printf("%s\n", (errorack == 0 ? "Error while sending ACK" : ""));
            }
        }
    }
}
Stop & Wait — Tricks to Remember
🔁 THE LOOP
While(!disconnect) calls sender() then receiver() alternately. They cooperate using the turn variable.
🎲 ERROR SIMULATION
rand() % 4 == 0 → 25% error chance. Non-zero = success. This simulates both frame errors AND ACK errors separately.
🔢 SEQUENCE LOGIC
Sender's p.seq increments each packet. Receiver's frexp (expected) also increments. If they match → new packet. If they don't → duplicate, re-ACK.
📝 EXAM Q: Why two error variables?
errorframe = error in the data packet going from sender to receiver. errorack = error in the ACK going back from receiver to sender. Both can fail independently.
Distance Vector Routing Routing Algorithm

Each router tells its neighbors what it knows. Based on Bellman-Ford algorithm. Each node maintains a routing table with distance and next-hop to every other node.

The Bellman-Ford Core Logic
THE KEY FORMULA
if (rt[i].dist[j] > dmat[i][k] + rt[k].dist[j])

Translation: "Can I reach j via intermediate node k cheaper than my current best path?" If yes → update!
1Initialize routing table from cost matrix (direct link costs)
2Triple nested loop: for every i (src) → j (dst) → k (intermediate)
3If shorter path found via k: update distance AND update "from" (next hop)
4Repeat until no updates (count == 0) → converged!
5Print final routing table for each node
DVR — Full Code
// DISTANCE VECTOR ROUTING — Networks Lab
#include <stdio.h>

struct node {
    unsigned dist[20];   // distance to each node
    unsigned from[20];   // next hop to reach each node
} rt[10];              // routing table for 10 nodes

int main() {
    int dmat[20][20];    // cost matrix
    int n, i, j, k, count = 0;

    printf("Enter the number of nodes: ");
    scanf("%d", &n);

    printf("Enter the cost matrix:\n");
    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++)
            scanf("%d", &dmat[i][j]);

    // Initialize routing table with direct link costs
    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++) {
            rt[i].dist[j] = dmat[i][j];
            rt[i].from[j] = j;  // initially, go directly
        }

    // Bellman-Ford: update until no change
    do {
        count = 0;
        for (i = 0; i < n; i++)
            for (j = 0; j < n; j++)
                for (k = 0; k < n; k++) {
                    // Can we reach j via k cheaper?
                    if (rt[i].dist[j] > dmat[i][k] + rt[k].dist[j]) {
                        rt[i].dist[j] = rt[i].dist[k] + rt[k].dist[j];
                        rt[i].from[j] = k;
                        count++;  // something changed
                    }
                }
    } while (count != 0);  // repeat till stable

    // Print routing tables
    for (i = 0; i < n; i++) {
        printf("\n\nState value for router %d\n", i + 1);
        for (j = 0; j < n; j++)
            printf("\n node %d via %d Distance %d",
                   j + 1, rt[i].from[j] + 1, rt[i].dist[j]);
    }
    printf("\n");
    return 0;
}
DVR — Tricks to Remember
🧠 THE MAGIC LINE
rt[i].dist[j] > dmat[i][k] + rt[k].dist[j]

Read as: "Current best distance from i→j" > "Cost of i→k (direct link)" + "k's best distance to j". If true → we found a shortcut via k!
🔄 WHY do-while?
We repeat until nothing changes (count == 0). This is convergence. Each iteration, more optimal paths propagate through the network. DVR is slow to converge — that's actually a known disadvantage!
📋 STRUCT TRICK
rt[i].dist[j] = shortest distance from node i to node j
rt[i].from[j] = which neighbor to use to get to j from i
💡 EXAM Q: What is the time complexity?
O(n³) per iteration × multiple iterations. Three nested loops over n nodes = O(n³). This is why DVR doesn't scale well to large networks.
Leaky Bucket Algorithm Congestion Control

Traffic shaping algorithm. Imagine a bucket with a hole — packets fill the bucket, but output rate is constant. Excess packets are dropped.

Visualization
Incoming packets (bursty)
→
BUCKET
size = buck_size
current = store
→
Outgoing (steady rate)
If incoming > (bucket_size - store) → DROP excess
Each round: store -= outgoing (constant drain)
Key Variables
buck_size = max bucket capacity
store = currently stored packets
incoming = new packets arriving
outgoing = constant drain rate
The Core Check
if (incoming <= buck_size - store)
→ fits in bucket → accept

else
→ drop excess, fill to max
Leaky Bucket — Full Code
// LEAKY BUCKET ALGORITHM — Networks Lab
#include <stdio.h>

int main() {
    int incoming, outgoing, buck_size, n, store = 0;

    printf("Enter bucket size, outgoing rate and no of inputs: ");
    scanf("%d%d%d", &buck_size, &outgoing, &n);

    while (n != 0) {
        printf("Enter the incoming packet size: ");
        scanf("%d", &incoming);
        printf("Incoming packet size %d\n", incoming);

        // Does it fit in bucket?
        if (incoming <= (buck_size - store)) {
            store += incoming;
            printf("Bucket buffer size %d out of %d\n", store, buck_size);
        } else {
            // Drop what doesn't fit
            printf("Dropped %d no of packets\n", incoming - (buck_size - store));
            printf("Bucket buffer size %d out of %d\n", buck_size, buck_size);
            store = buck_size;  // bucket is full
        }

        // Drain at constant outgoing rate
        store = store - outgoing;
        printf("After outgoing %d packets left out of %d in buffer\n",
               store, buck_size);
        n--;
    }
    return 0;
}
Leaky Bucket — Tricks
💧 THE ANALOGY
Bucket = buffer. Water in = incoming packets. Hole = outgoing rate. If too much water → overflow (drop packets). Hole always drains at same rate regardless of input.
🔢 DROP CALCULATION
Dropped = incoming - (buck_size - store). This is: "how many packets exceeded available space?"
📝 EXAM Q: Leaky vs Token Bucket
Leaky = fixed output rate always (strict). Token Bucket = can burst occasionally (flexible). Leaky drops excess; Token stores tokens to allow bursts.
⚠️ EDGE CASE
After outgoing, if store goes negative — in real code you'd clamp it to 0. The lab code assumes outgoing ≤ store always.
File Transfer Protocol TCP-Based

FTP is TCP + file I/O. Server reads a file and sends its contents. Client receives it and writes to a new file. The socket setup is identical to TCP — only the data handling changes.

FTP Flow
1Server: socket → bind → listen → accept (same as TCP)
2Client: socket → connect (same as TCP)
3Client sends filename to server via send()
4Server opens that file with fopen(), reads line by line
5Server sends each line; Client receives and writes to new file
6Server sends "completed" when done; Client detects it
🔑 KEY INSIGHT
FTP = TCP socket code + File I/O (fopen, fgets, fputs). If you know TCP, you know 70% of FTP already!
FTP Server SERVER
// FTP SERVER — Networks Lab
#include <stdio.h>
#include <stdlib.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <string.h>

int main() {
    int sd, newsd, n, port;
    char name[100], fileread[100];
    FILE *fp;
    struct sockaddr_in serveraddr, clientaddr;
    socklen_t len = sizeof(clientaddr);

    printf("Enter the port: ");
    scanf("%d", &port);

    // TCP socket setup (same as TCP server!)
    sd = socket(AF_INET, SOCK_STREAM, 0);
    if (sd < 0)
        printf("Can't create socket\n");

    serveraddr.sin_family = AF_INET;
    serveraddr.sin_addr.s_addr = INADDR_ANY;
    serveraddr.sin_port = htons(port);

    if (bind(sd, (struct sockaddr*)&serveraddr, sizeof(serveraddr)) < 0)
        printf("Can't bind\n");

    listen(sd, 5);

    newsd = accept(sd, (struct sockaddr*)&clientaddr, &len);
    if (newsd < 0)
        printf("Can't accept\n");

    // Receive filename from client
    n = recv(newsd, name, 100, 0);
    name[n] = '\0';
    printf("File requested: %s\n", name);

    // Open the file
    fp = fopen(name, "r");
    if (fp == NULL) {
        send(newsd, "File not found", 100, 0);
    } else {
        // Send file line by line
        while (fgets(fileread, sizeof(fileread), fp)) {
            send(newsd, fileread, sizeof(fileread), 0);
            sleep(1);
        }
        // Signal completion
        send(newsd, "completed", 100, 0);
        fclose(fp);
    }

    close(newsd);
    close(sd);
    return 0;
}
FTP Client CLIENT
// FTP CLIENT — Networks Lab
#include <stdio.h>
#include <stdlib.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <string.h>

int main() {
    FILE *fp;
    int csd, port, n;
    char name[100], rcv[100], newname[100];
    struct sockaddr_in serveraddr;

    printf("Enter the port: ");
    scanf("%d", &port);

    // TCP socket setup (same as TCP client!)
    csd = socket(AF_INET, SOCK_STREAM, 0);
    if (csd < 0)
        printf("Can't create socket\n");

    serveraddr.sin_family = AF_INET;
    serveraddr.sin_addr.s_addr = INADDR_ANY;
    serveraddr.sin_port = htons(port);

    if (connect(csd, (struct sockaddr*)&serveraddr, sizeof(serveraddr)) < 0)
        printf("Can't connect\n");

    // Send requested filename to server
    printf("Enter filename to request: ");
    scanf("%s", name);
    send(csd, name, sizeof(name), 0);

    // Enter new filename to save as
    printf("Enter new filename to save: ");
    scanf("%s", newname);
    fp = fopen(newname, "w");

    // Receive and write until "completed"
    while ((n = recv(csd, rcv, 100, 0)) > 0) {
        if (strcmp(rcv, "completed") == 0) {
            printf("File transfer completed!\n");
            break;
        }
        fputs(rcv, fp);
    }

    fclose(fp);
    close(csd);
    return 0;
}
FTP — Copy-Paste Tricks
Server → Client Changes
✅ COPY ENTIRE TCP server/client boilerplate
Socket setup, bind, listen, accept (server) / connect (client) — 100% identical to TCP. Don't rewrite this!
➕ ADD: File handling
Server: fopen(name, "r") + fgets() loop + send("completed")
Client: fopen(newname, "w") + recv loop + strcmp("completed") check + fputs()
💡 TERMINATION TRICK
Server sends "completed" as a magic string to signal end of file. Client checks strcmp(rcv, "completed") == 0 to break the loop. This is the "EOF signal" over the network.
📋 FTP vs TCP diff in one sentence
FTP = TCP + fopen/fgets/fputs/fclose. Everything else is the same.
